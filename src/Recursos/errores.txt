public static boolean validarString(String palabra) {
        // Expresión regular para permitir solo letras, incluyendo acentos, ñ y ç.
        String patron = "^[a-zA-ZáéíóúñÁÉÍÓÚÇç]+$";

        return palabra.matches(patron);
    }
